function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const p = e.parameter; const callback = p.callback || 'callback'; try { // 1. טיפול בפעולות שמירה (עדכון/הוספה/מחיקה) if (p.action && p.action !== 'load') { if (p.action === 'update') { const colMap = { 'date': 1, 'day': 2, 'time': 3, 'teacher': 4, 'type': 5, 'loc': 6 }; const col = colMap[p.field]; if (col && p.rowIndex) { sheet.getRange(parseInt(p.rowIndex), col).setValue(p.value); } } else if (p.action === 'addRow') { // מציאת השורה האמיתית האחרונה (מתעלם מתאים שרק מעוצבים עם צבע בלי טקסט) const data = sheet.getDataRange().getValues(); let realLastRow = data.length; while (realLastRow > 0 && data[realLastRow - 1].join('').trim() === '') { realLastRow--; } const newRow = realLastRow + 1; // סדר עמודות: תאריך(1), יום(2), שעה(3), מורה(4), סוג(5), מיקום(6), מקסימום(7) sheet.getRange(newRow, 1, 1, 7).setValues([[p.date || '', p.day || '', p.time || '', p.teacher || '', p.type || '', p.loc || 'סטודיו', 7]]); } else if (p.action === 'deleteRow') { if (p.rowIndex) sheet.deleteRow(parseInt(p.rowIndex)); } // החזרת אישור שהשמירה הצליחה return ContentService.createTextOutput(callback + '(' + JSON.stringify({success: true}) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } // 2. פעולת קריאה (load) - מחזיר את כל הנתונים לתצוגה באתר const fullData = sheet.getDataRange().getValues(); const result = fullData.slice(1).map((r, i) => ({ rowIndex: i + 2, date: formatDate(r[0]), day: String(r[1] || ''), time: String(r[2] || '').trim(), teacher: r[3] ? String(r[3]).trim() : '', type: String(r[4] || '').trim(), loc: r[5] ? String(r[5]).trim() : '', max: Number(r[6]) || 7 })).filter(r => r.date || r.type || r.teacher); return ContentService.createTextOutput(callback + '(' + JSON.stringify(result) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } catch (err) { return ContentService.createTextOutput(callback + '(' + JSON.stringify({success: false, error: err.toString()}) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } } // גיבוי למקרה שהדפדפן שולח בטעות בקשת POST (למרות שהאתר שלנו מכוון ל-GET) function doPost(e) { return doGet(e); } function formatDate(val) { if (!val) return ''; if (val instanceof Date) { return String(val.getDate()).padStart(2, '0') + '/' + String(val.getMonth() + 1).padStart(2, '0'); } return String(val).split(' ')[0]; }
