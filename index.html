function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const p = e.parameter; const callback = p.callback || 'callback'; try { // 1. טיפול בפעולות שמירה (עדכון/הוספה/מחיקה) if (p.action && p.action !== 'load') { if (p.action === 'update') { // מיפוי עמודות לפי המבנה שסיפקת: תאריך=1, יום=2, שעה=3, מורה=4, סוג=5, מיקום=6 const colMap = { 'date': 1, 'day': 2, 'time': 3, 'teacher': 4, 'type': 5, 'loc': 6 }; const col = colMap[p.field]; if (col && p.rowIndex) { sheet.getRange(parseInt(p.rowIndex), col).setValue(p.value); } } else if (p.action === 'addRow') { const data = sheet.getDataRange().getValues(); let realLastRow = data.length; while (realLastRow > 0 && data[realLastRow - 1].join('').trim() === '') realLastRow--; // הוספת שורה חדשה (עמודה 7 היא מקסימום משתתפים ברירת מחדל 7) sheet.getRange(realLastRow + 1, 1, 1, 7).setValues([[p.date || '', p.day || '', p.time || '', p.teacher || '', p.type || '', p.loc || 'סטודיו', 7]]); } else if (p.action === 'deleteRow') { if (p.rowIndex) sheet.deleteRow(parseInt(p.rowIndex)); } // --- ביצוע מיון אוטומטי לאחר כל שינוי --- // ממיין משורה 2 (כדי לדלג על הכותרות) עד השורה האחרונה const lastRow = sheet.getLastRow(); const lastCol = sheet.getLastColumn(); if (lastRow > 1) { const rangeToSort = sheet.getRange(2, 1, lastRow - 1, lastCol); rangeToSort.sort([ {column: 1, ascending: true}, // מיון ראשון לפי תאריך {column: 3, ascending: true} // מיון שני לפי שעה ]); } // שמירה פיזית מיידית SpreadsheetApp.flush(); return ContentService.createTextOutput(callback + '(' + JSON.stringify({success: true}) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } // 2. פעולת קריאה (load) const fullData = sheet.getDataRange().getValues(); const result = fullData.slice(1).map((r, i) => ({ rowIndex: i + 2, date: formatDate(r[0]), day: String(r[1] || ''), time: String(r[2] || '').trim(), teacher: r[3] ? String(r[3]).trim() : '', type: String(r[4] || '').trim(), loc: r[5] ? String(r[5]).trim() : '', max: Number(r[6]) || 7 })).filter(r => r.date || r.type || r.teacher); return ContentService.createTextOutput(callback + '(' + JSON.stringify(result) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } catch (err) { return ContentService.createTextOutput(callback + '(' + JSON.stringify({success: false, error: err.toString()}) + ')').setMimeType(ContentService.MimeType.JAVASCRIPT); } } function doPost(e) { return doGet(e); } function formatDate(val) { if (!val) return ''; if (val instanceof Date) { return ('0' + val.getDate()).slice(-2) + '/' + ('0' + (val.getMonth() + 1)).slice(-2); } const str = String(val).trim(); // אם זה פורמט DD/MM/YYYY או דומה, נחלץ רק DD/MM const match = str.match(/(\d{1,2})[\/\-\.](\d{1,2})/); if (match) { return ('0' + match[1]).slice(-2) + '/' + ('0' + match[2]).slice(-2); } return str; }
